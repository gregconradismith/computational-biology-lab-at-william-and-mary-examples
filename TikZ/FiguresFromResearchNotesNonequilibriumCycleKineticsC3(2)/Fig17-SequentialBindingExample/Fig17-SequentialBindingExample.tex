\documentclass[tikz,border=10pt]{standalone}
\usepackage{pgfplots}
\pgfplotsset{compat=1.17}
\input{../../definitions}

\begin{document}
\pgfplotsset{every axis/.style={
clip=true,
height=8cm,
width=10cm,
xmax=100,
xmin=0.01,
ymin=0,
ymax=1.1,
axis x line=bottom,
legend pos=north west,
xlabel={$x$},
axis y line=left,
ylabel near ticks,
xlabel near ticks,
ylabel style={rotate=-90},
domain=0.001:100,
samples=200]}}

\begin{tikzpicture}

    % three-state sequential binding model (Equation 3.1 in Allostery paper)
    \def\Za{1}
    \def\Zb{\Kappabstar*x}
    \def\Zc{\Kappabstar*\Kappacstar*x^2}
    \def\Ztot{(\Za+\Zb+\Zc)} % ()'s are important!

    % these parameters may be changed
    \def\KappabstarBlue{0.1}
    \def\KappacstarBlue{1}
    \def\KappabstarRed{0.5}
    \def\KappacstarRed{1}

    \begin{semilogxaxis}[ylabel={$\pi_c$}]

        \def\Kappabstar{\KappabstarBlue}
        \def\Kappacstar{\KappacstarBlue}
        \addplot[blue] {\Zc/\Ztot};
        \addlegendentry{$\kappabstar = \KappabstarBlue,  \kappacstar = \KappacstarBlue$}

        \def\Kappabstar{\KappabstarRed}
        \def\Kappacstar{\KappacstarRed}
        \addplot[red] {\Zc/\Ztot};
        \addlegendentry{$\kappabstar = \KappabstarRed,  \kappacstar = \KappacstarRed$}

    \end{semilogxaxis}

\end{tikzpicture}

\end{document}
